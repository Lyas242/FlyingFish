## 一、测试背景

本文档用于指导如何使用存储过程生成测试数据，并验证在幂次方扩容前后，通过**订单号**和**用户ID**查询订单是否正常。

### 1.1 扩容场景

- **扩容前**：2库 × 4表 = 8个物理分片
- **扩容后**：2库 × 8表 = 16个物理分片

### 1.2 测试重点：库内迁移

本次测试重点验证**库内迁移**场景，即数据在同一个数据库内从旧表迁移到新表，例如：
- `damai_order_0.d_order_0` → `damai_order_0.d_order_4`

这样可以简化测试，避免跨库迁移的复杂性，同时也能充分验证幂次方扩容的核心逻辑。

## 二、测试用户ID选择策略

### 2.1 扩容前（2库×4表）路由规则

```
基因位数：3位（1位库 + 2位表）
分表：orderNumber & 0b11  （取后2位）
分库：(orderNumber >> 2) & 0b1  （取第3位）

示例：
userId = 1004 = 0b1111101100
├─ 表索引 = 1004 & 0b11 = 0b00 = 0 → d_order_0
└─ 库索引 = (1004 >> 2) & 0b1 = 0b11111011 & 0b1 = 0 → damai_order_0
结果：damai_order_0.d_order_0
```

### 2.2 推荐测试用户ID（库内迁移场景）

| 用户ID | 二进制(后4位) | 扩容前库索引 | 扩容前表索引 | 扩容前位置 | 扩容后库索引 | 扩容后表索引 | 扩容后位置 | 迁移类型 |
|--------|--------------|-------------|-------------|-----------|-------------|-------------|-----------|----------|
| **1004** | 0100 | 0 | 0 | damai_order_0.d_order_0 | 0 | 4 | damai_order_0.d_order_4 | ✅ **库内迁移** |
| **1005** | 0101 | 0 | 1 | damai_order_0.d_order_1 | 0 | 5 | damai_order_0.d_order_5 | ✅ **库内迁移** |
| **1006** | 0110 | 0 | 2 | damai_order_0.d_order_2 | 0 | 6 | damai_order_0.d_order_6 | ✅ **库内迁移** |
| **1007** | 0111 | 0 | 3 | damai_order_0.d_order_3 | 0 | 7 | damai_order_0.d_order_7 | ✅ **库内迁移** |

**选择原因**：
1. 这4个用户ID的**第4位（库位）为0**，确保始终路由到 `damai_order_0`
2. 这4个用户ID的**第3位（新增表位）为1**，确保扩容时会触发迁移
3. 扩容前后都在同一个库内，简化测试验证

### 2.3 扩容后（2库×8表）路由规则

```
基因位数：4位（1位库 + 3位表）
分表：orderNumber & 0b111  （取后3位）
分库：(orderNumber >> 3) & 0b1  （取第4位）

示例：
userId = 1004 = 0b1111101100
├─ 表索引 = 1004 & 0b111 = 0b100 = 4 → d_order_4
└─ 库索引 = (1004 >> 3) & 0b1 = 0b1111101 & 0b1 = 0 → damai_order_0
结果：damai_order_0.d_order_4
```

**扩容前后路由对比**：

| 用户ID | 扩容前基因(3位) | 扩容前位置 | 扩容后基因(4位) | 扩容后位置 | 迁移说明 |
|--------|----------------|-----------|----------------|-----------|----------|
| 1004 | 0b000 | damai_order_0.d_order_0 | 0b0100 | damai_order_0.d_order_4 | ✅ 库内迁移 |
| 1005 | 0b001 | damai_order_0.d_order_1 | 0b0101 | damai_order_0.d_order_5 | ✅ 库内迁移 |
| 1006 | 0b010 | damai_order_0.d_order_2 | 0b0110 | damai_order_0.d_order_6 | ✅ 库内迁移 |
| 1007 | 0b011 | damai_order_0.d_order_3 | 0b0111 | damai_order_0.d_order_7 | ✅ 库内迁移 |

## 三、一键生成测试数据存储过程（库内迁移版）

### 3.1 核心存储过程（只需执行一次）

**特点**：
- ✅ 只需在`damai_order_0`库执行
- ✅ 自动生成4个测试用户的数据（1004-1007）
- ✅ 这4个用户的数据都在`damai_order_0`库内
- ✅ 扩容后数据从`d_order_0-3`迁移到`d_order_4-7`（库内迁移）
- ✅ 一条命令搞定，无需手动指定用户ID

```sql
-- ========================================
-- 一键生成幂次方扩容测试数据（库内迁移版）
-- 使用方法：
--   1. 在 damai_order_0 库执行本脚本
--   2. 执行：CALL generate_expansion_test_data();
--   3. 完成！无需其他操作
-- 
-- 测试特点：
--   - 所有测试数据都在 damai_order_0 库内
--   - 扩容时数据会在库内从旧表迁移到新表
--   - 例如：d_order_0 → d_order_4
-- ========================================

DELIMITER $$

DROP PROCEDURE IF EXISTS generate_expansion_test_data$$

CREATE PROCEDURE generate_expansion_test_data()
BEGIN
    DECLARE v_counter INT;
    DECLARE v_order_id BIGINT;
    DECLARE v_order_number BIGINT;
    DECLARE v_current_time DATETIME;
    DECLARE v_base_timestamp BIGINT DEFAULT 1288834974657;
    DECLARE v_sequence INT;
    DECLARE v_datacenter_id INT DEFAULT 1;
    DECLARE v_worker_id INT DEFAULT 1;
    DECLARE v_program_id BIGINT DEFAULT 1;  -- 默认节目ID
    DECLARE v_orders_per_user INT DEFAULT 5;  -- 每个用户生成5条订单
    
    -- 基因法相关变量（扩容前：2库4表）
    DECLARE v_table_gene_length INT DEFAULT 2;  -- log2(4) = 2位
    DECLARE v_database_gene_length INT DEFAULT 1; -- log2(2) = 1位
    DECLARE v_total_gene_length INT DEFAULT 3;  -- 总基因位数
    DECLARE v_gene_mask BIGINT;
    DECLARE v_user_gene BIGINT;
    DECLARE v_table_index INT;
    DECLARE v_database_index INT;
    DECLARE v_target_table VARCHAR(50);
    DECLARE v_insert_sql TEXT;
    
    -- 4个测试用户ID（库内迁移场景）
    DECLARE v_user_id BIGINT;
    DECLARE v_user_idx INT DEFAULT 0;
    
    SET v_current_time = NOW();
    SET v_gene_mask = (1 << v_total_gene_length) - 1;  -- 0b111 = 7
    
    -- 创建临时表存储测试用户ID
    DROP TEMPORARY TABLE IF EXISTS temp_test_users;
    CREATE TEMPORARY TABLE temp_test_users (
        idx INT AUTO_INCREMENT PRIMARY KEY,
        user_id BIGINT,
        user_gene INT COMMENT '扩容前3位基因',
        user_gene_4bit INT COMMENT '扩容后4位基因',
        table_index INT COMMENT '扩容前表索引',
        database_index INT COMMENT '扩容前库索引',
        database_name VARCHAR(50),
        table_name VARCHAR(50),
        target_table_index INT COMMENT '扩容后表索引',
        target_table_name VARCHAR(50) COMMENT '扩容后表名',
        orders_created INT DEFAULT 0
    );
    
    -- 插入4个测试用户的路由信息（库内迁移场景）
    -- 用户ID后4位分别为：0100, 0101, 0110, 0111
    -- 扩容前：库索引=0，表索引=0,1,2,3 → damai_order_0.d_order_{0,1,2,3}
    -- 扩容后：库索引=0，表索引=4,5,6,7 → damai_order_0.d_order_{4,5,6,7}
    INSERT INTO temp_test_users (user_id, user_gene, user_gene_4bit, table_index, database_index, database_name, table_name, target_table_index, target_table_name)
    VALUES 
        (1004, 0, 4, 0, 0, 'damai_order_0', 'd_order_0', 4, 'd_order_4'),
        (1005, 1, 5, 1, 0, 'damai_order_0', 'd_order_1', 5, 'd_order_5'),
        (1006, 2, 6, 2, 0, 'damai_order_0', 'd_order_2', 6, 'd_order_6'),
        (1007, 3, 7, 3, 0, 'damai_order_0', 'd_order_3', 7, 'd_order_7');
    
    -- 为每个用户生成订单
    SET v_user_idx = 0;
    WHILE v_user_idx < 4 DO
        SET v_user_idx = v_user_idx + 1;
        
        -- 获取当前用户信息
        SELECT user_id, user_gene, table_index, database_index, database_name, table_name
        INTO v_user_id, v_user_gene, v_table_index, v_database_index, @v_db_name, @v_table_name
        FROM temp_test_users WHERE idx = v_user_idx;
        
        -- 为当前用户生成订单
        SET v_counter = 1;
        WHILE v_counter <= v_orders_per_user DO
            -- 生成订单ID
            SET v_order_id = (UNIX_TIMESTAMP(v_current_time) * 1000000) + v_user_id * 100 + v_counter;
            
            -- 生成订单号（包含基因）
            SET v_sequence = v_user_idx * 100 + v_counter;
            SET v_order_number = (
                ((UNIX_TIMESTAMP(v_current_time) * 1000 - v_base_timestamp) << 22)
                | (v_datacenter_id << 17)
                | (v_worker_id << 12)
                | (v_sequence << v_total_gene_length)
                | v_user_gene  -- ⭐ 关键：低3位是用户基因
            );
            
            -- 动态构建插入SQL（只在damai_order_0库内）
            SET @insert_sql = CONCAT(
                'INSERT INTO ', @v_db_name, '.', @v_table_name, ' ',
                '(id, order_number, identifier_id, program_id, program_item_picture, user_id, ',
                'program_title, program_place, program_show_time, program_permit_choose_seat, ',
                'distribution_mode, take_ticket_mode, order_price, pay_order_type, ',
                'order_status, reconciliation_status, order_version, create_order_time, create_time, status) ',
                'VALUES (',
                v_order_id, ', ',
                v_order_number, ', ',
                v_order_id, ', ',
                v_program_id, ', ',
                '''https://example.com/program.jpg'', ',
                v_user_id, ', ',
                '''测试节目-用户', v_user_id, '-订单', v_counter, ''', ',
                '''测试场馆'', ',
                '''', DATE_ADD(v_current_time, INTERVAL 30 DAY), ''', ',
                '1, ',  -- program_permit_choose_seat
                '''电子票'', ',
                '''取票方式：凭身份证入场'', ',
                100.00 * v_counter, ', ',
                '1, ',  -- pay_order_type
                '1, ',  -- order_status
                '1, ',  -- reconciliation_status
                '3, ',  -- order_version
                '''', v_current_time, ''', ',
                '''', v_current_time, ''', ',
                '1)'    -- status
            );
            
            -- 执行插入
            PREPARE stmt FROM @insert_sql;
            EXECUTE stmt;
            DEALLOCATE PREPARE stmt;
            
            SET v_counter = v_counter + 1;
        END WHILE;
        
        -- 更新已创建的订单数
        UPDATE temp_test_users SET orders_created = v_orders_per_user WHERE idx = v_user_idx;
        
    END WHILE;
    
    -- 输出测试用户信息
    SELECT 
        '✅ 测试数据生成完成！' AS message,
        '请使用以下用户ID进行库内迁移测试' AS note;
    
    SELECT 
        user_id AS '推荐测试用户ID',
        CONCAT('0b', LPAD(BIN(user_gene), 3, '0')) AS '扩容前基因(3位)',
        CONCAT('0b', LPAD(BIN(user_gene_4bit), 4, '0')) AS '扩容后基因(4位)',
        CONCAT(database_name, '.', table_name) AS '扩容前位置',
        CONCAT(database_name, '.', target_table_name) AS '扩容后位置',
        orders_created AS '订单数量',
        '✅ 库内迁移' AS '迁移类型'
    FROM temp_test_users
    ORDER BY user_id;
    
    -- 输出汇总统计
    SELECT 
        '数据生成汇总' AS '统计项',
        4 AS '测试用户数',
        5 AS '每用户订单数',
        20 AS '总订单数',
        'damai_order_0库内' AS '数据所在位置',
        '库内迁移（d_order_0-3 → d_order_4-7）' AS '扩容方式';
    
    -- 清理临时表
    DROP TEMPORARY TABLE IF EXISTS temp_test_users;
    
END$$

DELIMITER ;
```

/**

        log.info("========== 测试1：扩容前通过用户ID查询 ==========");
        
        for (Long userId : TEST_USER_IDS) {

                Wrappers.lambdaQuery(Order.class).eq(Order::getUserId, userId)

            log.info("用户ID: {}, 订单数: {}", userId, orders.size());

            List<Long> orderNumbers = orders.stream()
                .map(Order::getOrderNumber)
                .collect(Collectors.toList());
            userOrderNumbersMap.put(userId, orderNumbers);
            
            // 验证订单号的基因位
            for (Order order : orders) {
                long orderGene = order.getOrderNumber() & 0b111;  // 取后3位
                long userGene = userId & 0b111;
                
                log.info("  订单号: {}, 订单基因: {}, 用户基因: {}, 匹配: {}", 
                    order.getOrderNumber(), orderGene, userGene, orderGene == userGene);
                
                Assertions.assertEquals(userGene, orderGene, 
                    String.format("订单号的基因位(%d)应该等于用户ID的基因位(%d)", orderGene, userGene));
            }
            
            Assertions.assertTrue(orders.size() > 0, "用户 " + userId + " 应该有订单数据");
        }

    public void test05_verifyDataDistribution() {
        log.info("========== 测试5：验证扩容后库内迁移数据分布 ==========");
        
        Map<Long, String> expectedLocations = Map.of(
            1004L, "damai_order_0.d_order_4 (从d_order_0库内迁移)",
            1005L, "damai_order_0.d_order_5 (从d_order_1库内迁移)",
            1006L, "damai_order_0.d_order_6 (从d_order_2库内迁移)",
            1007L, "damai_order_0.d_order_7 (从d_order_3库内迁移)"
        );
        
        for (Long userId : TEST_USER_IDS) {
            List<Order> orders = orderMapper.selectList(
                Wrappers.lambdaQuery(Order.class).eq(Order::getUserId, userId)
            );
            
            if (!orders.isEmpty()) {
                Order order = orders.get(0);
                long orderNumber = order.getOrderNumber();
                
                // 扩容后：4位基因（1位库 + 3位表）
                long tableIndex = orderNumber & 0b111;  // 取后3位
                long dbIndex = (orderNumber >> 3) & 0b1;  // 取第4位
                
                String actualLocation = String.format("damai_order_%d.d_order_%d", dbIndex, tableIndex);
                
                log.info("用户ID: {}, 预期位置: {}, 实际路由: {}", 
                    userId, expectedLocations.get(userId), actualLocation);
                
                // 验证所有用户都在 damai_order_0 库内
                Assertions.assertEquals(0, dbIndex, 
                    "用户 " + userId + " 应该在 damai_order_0 库内");
                
                // 验证表索引是 4-7
                Assertions.assertTrue(tableIndex >= 4 && tableIndex <= 7,
                    "用户 " + userId + " 应该在 d_order_4-7 表中");
            }
        }
        
        log.info("✅ 库内迁移数据分布验证完成");
    }
